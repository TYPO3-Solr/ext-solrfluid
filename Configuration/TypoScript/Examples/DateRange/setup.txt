# Enable a date range facet on created field

# Enable suggest
plugin.tx_solr.search.faceting.facets {
    created {
        label = Created Between
        field = created

        type = dateRange

        # these are the default values, you do not need to specify a dateRange configuration block if the defaults work for you
        dateRange {
            start = NOW/DAY-1YEAR
            end = NOW/DAY+1YEAR
            gap = +1DAY
        }
    }
}

    solr-datepicker = EXT:solr/Resources/JavaScript/JQuery/jquery-ui.datepicker.min.js

    solr-daterange = EXT:solrfluid/Resources/Public/JavaScript/facet_daterange.js
}
